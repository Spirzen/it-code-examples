@echo off
setlocal EnableExtensions
set "SRC=C:\Projects\app"
set "DST=D:\Backup\app"

if not exist "%SRC%" (
  echo Source not found: %SRC%
  exit /b 1
)

robocopy "%SRC%" "%DST%" /MIR /R:2 /W:5 /NFL /NDL /NP
set "RC=%ERRORLEVEL%"
if %RC% GEQ 8 (
  echo Backup failed, robocopy code %RC%
  exit /b 1
)
echo Backup OK
exit /b 0
